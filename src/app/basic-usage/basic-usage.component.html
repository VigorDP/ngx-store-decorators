Basic usage

Decorator Select (for handle Observables):
@Select(CounterFacadeService, 'count$')
public count$: Observable{{'<'}}number>; => {{ count$ | async }} | async;
  
@Select(CounterFacadeService, 'countMultiplyBy$', {{'{'}}args: [3]})
public countMultiplyBy$: Observable{{'<'}}number>; => {{ countMultiplyBy$ | async }} | async;
Decorator Subscribe (for handle Observables):
@Subscribe(CounterFacadeService, 'count$')
public count: number; => {{ count }};
  
@Subscribe(CounterFacadeService, 'countMultiplyBy$', {{'{'}}args: [3]})
public countMultiplyBy: number; => {{ countMultiplyBy }}
Decorator Get (for handle any values):
@Get(CounterFacadeService, 'count')
public getCount: number; => {{ getCount }}
  
@Get(CounterFacadeService, 'countFunction')
public getCountFunction: number; => {{ getCountFunction }}
  
@Get(CounterFacadeService, 'countFunctionWithArg', {{'{'}}args: [3]})
public getCountFunctionWithArguments: number; => {{ getCountFunctionWithArguments }}
